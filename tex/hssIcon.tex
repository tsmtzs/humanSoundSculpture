\documentclass[tikz]{standalone}
\usepackage{xcolor}

\usetikzlibrary{calc}

\pagestyle{empty}

% Background color
\definecolor{background}{HTML}{293F50}

% Tikz background rectangle side length
\def\outerSide{4.05cm}
% Center rectangle side length
\def\centerDeviation{0.95cm}

\begin{document}
\pagecolor{background}
\centering
\begin{tikzpicture}
  \coordinate (center) at ($0.5*(\outerSide,\outerSide)$);
  \coordinate (deviation) at (\centerDeviation,\centerDeviation);
  % Background.
  \fill [color=background] (0,0) rectangle (\outerSide,\outerSide);
  %
  \fill [color=gray!35, rounded corners=2.4cm]
  ($(center) - (deviation)$) rectangle ($(center) + (deviation)$);
\end{tikzpicture}
\end{document}
